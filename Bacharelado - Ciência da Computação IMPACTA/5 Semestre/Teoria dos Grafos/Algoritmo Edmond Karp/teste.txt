def bfs(capacidade, fluxo, origem, destino, pai, adjacencia):
    """
    BFS para encontrar caminho aumentante.
    Usa lista de adjacência para evitar iterar sobre todos os nós.
    """
    n = len(capacidade)
    visitado = [False] * n
    fila = deque()

    fila.append(origem)
    visitado[origem] = True
    pai[origem] = -1

    while fila:
        u = fila.popleft()

        # Só percorre vizinhos reais (não todos os n nós)
        for v in adjacencia[u]:
            if not visitado[v] and capacidade[u][v] - fluxo[u][v] > 0:
                fila.append(v)
                pai[v] = u
                visitado[v] = True

                if v == destino:
                    return True  # Early exit

    return False

def construir_adjacencia(capacidade):
    """Constrói lista de adjacência (incluindo arestas reversas para fluxo)."""
    n = len(capacidade)
    adj = [[] for _ in range(n)]
    for u in range(n):
        for v in range(n):
            if capacidade[u][v] > 0:
                adj[u].append(v)
                if u not in adj[v]:  # aresta reversa (para cancelamento de fluxo)
                    adj[v].append(u)
    return adj


def edmonds_karp_steps(capacidade, origem, destino):
    """
    Executa Edmonds-Karp e retorna:
    - fluxo máximo
    - matriz de fluxo final
    - lista de steps para animação
    """
    n = len(capacidade)
    fluxo = [[0] * n for _ in range(n)]
    pai = [-1] * n
    fluxo_maximo = 0
    adjacencia = construir_adjacencia(capacidade)

    steps = []

    while bfs(capacidade, fluxo, origem, destino, pai, adjacencia):
        # Reconstruir caminho e calcular gargalo
        caminho = []
        gargalo = float('inf')
        v = destino

        while v != origem:
            u = pai[v]
            caminho.append((u, v))
            gargalo = min(gargalo, capacidade[u][v] - fluxo[u][v])
            v = u

        caminho.reverse()

        # Salvar snapshot ANTES de atualizar o fluxo
        steps.append({
            "caminho": caminho,
            "gargalo": gargalo,
            "fluxo_antes": [row[:] for row in fluxo],
            "fluxo_acumulado": fluxo_maximo
        })

        # Atualizar fluxo
        v = destino
        while v != origem:
            u = pai[v]
            fluxo[u][v] += gargalo
            fluxo[v][u] -= gargalo
            v = u

        fluxo_maximo += gargalo

    return fluxo_maximo, fluxo, steps

def plot_step(capacidade, fluxo, caminho=None, titulo="", node_labels=None):
    """
    Plota o grafo com:
    - Arestas do caminho aumentante em vermelho e mais grossas
    - Arestas saturadas em cinza tracejado
    - Nós coloridos (origem=verde, destino=vermelho, intermediários=azul claro)
    - Labels legíveis nos nós
    """
    G = nx.DiGraph()
    n = len(capacidade)

    for u in range(n):
        for v in range(n):
            if capacidade[u][v] > 0:
                G.add_edge(u, v)

    pos = {
        0: (-1.5, 0),
        1: (0, 1.2),
        2: (0, -1.2),
        3: (2, 1.2),
        4: (2, -1.2),
        5: (3.5, 0),
    }

    # Labels dos nós
    labels = node_labels if node_labels else {i: str(i) for i in range(n)}

    # Cores dos nós
    node_colors = []
    for i in range(n):
        if i == 0:
            node_colors.append("#2ecc71")   # verde - origem
        elif i == n - 1:
            node_colors.append("#e74c3c")   # vermelho - destino
        else:
            node_colors.append("#85c1e9")   # azul claro - intermediário

    # Classificar arestas
    edge_colors = []
    edge_widths = []
    edge_styles = []
    caminho_set = set(caminho) if caminho else set()

    for u, v in G.edges():
        if (u, v) in caminho_set:
            edge_colors.append("#e74c3c")   # vermelho
            edge_widths.append(3.0)
            edge_styles.append("solid")
        elif fluxo[u][v] >= capacidade[u][v] and capacidade[u][v] > 0:
            edge_colors.append("#bdc3c7")   # cinza - saturada
            edge_widths.append(1.5)
            edge_styles.append("dashed")
        else:
            edge_colors.append("#2c3e50")   # escuro normal
            edge_widths.append(1.5)
            edge_styles.append("solid")

    fig, ax = plt.subplots(figsize=(10, 6))

    # Desenhar nós
    nx.draw_networkx_nodes(G, pos, node_color=node_colors, node_size=700,
                           edgecolors="black", linewidths=1.5, ax=ax)
    nx.draw_networkx_labels(G, pos, labels=labels, font_size=14,
                            font_weight="bold", ax=ax)

    # Desenhar arestas (uma a uma para estilos diferentes)
    for idx, (u, v) in enumerate(G.edges()):
        nx.draw_networkx_edges(
            G, pos,
            edgelist=[(u, v)],
            edge_color=[edge_colors[idx]],
            width=edge_widths[idx],
            style=edge_styles[idx],
            arrows=True,
            arrowsize=20,
            connectionstyle="arc3,rad=0.1",
            ax=ax
        )

    # Labels das arestas: fluxo/capacidade
    edge_labels = {}
    for u, v in G.edges():
        f = max(fluxo[u][v], 0)  # Não mostrar fluxo negativo
        c = capacidade[u][v]
        edge_labels[(u, v)] = f"{f}/{c}"

    nx.draw_networkx_edge_labels(
        G, pos, edge_labels=edge_labels,
        font_size=10, font_color="#2c3e50",
        bbox=dict(boxstyle="round,pad=0.2", fc="white", ec="gray", alpha=0.8),
        ax=ax
    )

    ax.set_title(titulo, fontsize=14, fontweight="bold", pad=15)
    ax.margins(0.15)
    ax.axis("off")
    plt.tight_layout()
    plt.show()

def animar(capacidade, steps, node_labels=None, intervalo=2.0):
    """
    Anima passo a passo o algoritmo.
    
    Args:
        capacidade: matriz de capacidades
        steps: lista de passos do algoritmo
        node_labels: dict com nomes dos nós
        intervalo: segundos entre cada passo
    """
    n = len(capacidade)
    fluxo_acumulado = [[0] * n for _ in range(n)]

    for i, step in enumerate(steps):
        clear_output(wait=True)

        caminho = step["caminho"]
        gargalo = step["gargalo"]
        fluxo_antes = step["fluxo_acumulado"]

        # Atualizar fluxo acumulado
        for u, v in caminho:
            fluxo_acumulado[u][v] += gargalo
            fluxo_acumulado[v][u] -= gargalo

        # Info textual
        caminho_str = " → ".join(
            [str(node_labels.get(u, u) if node_labels else u) for u, v in caminho]
            + [str(node_labels.get(caminho[-1][1], caminho[-1][1]) if node_labels else caminho[-1][1])]
        )

        print(f"{'='*50}")
        print(f"  Iteração {i+1}/{len(steps)}")
        print(f"  Caminho: {caminho_str}")
        print(f"  Gargalo: {gargalo}")
        print(f"  Fluxo acumulado: {fluxo_antes} → {fluxo_antes + gargalo}")
        print(f"{'='*50}")

        plot_step(
            capacidade, fluxo_acumulado, caminho,
            titulo=f"Iteração {i+1} — Gargalo: {gargalo} | Caminho: {caminho_str}",
            node_labels=node_labels
        )

        if i < len(steps) - 1:
            time.sleep(intervalo)

    # Estado final
    clear_output(wait=True)
    fluxo_max = sum(fluxo_acumulado[0])
    print(f"\n{'='*50}")
    print(f"  ✅ FLUXO MÁXIMO ENCONTRADO: {fluxo_max}")
    print(f"  Total de iterações: {len(steps)}")
    print(f"{'='*50}\n")

    plot_step(
        capacidade, fluxo_acumulado, caminho=None,
        titulo=f"Estado Final — Fluxo Máximo: {fluxo_max}",
        node_labels=node_labels
    )
